let currentDate = new Date();
let data = JSON.parse(localStorage.getItem("roseFinanceData")) || {};
let selectedDay = null;

const calendar = document.getElementById("calendar");
const monthYear = document.getElementById("monthYear");
const modal = document.getElementById("entryModal");
const selectedDateText = document.getElementById("selectedDate");
const entryList = document.getElementById("entryList");

const incomeInput = document.getElementById("incomeInput");
const expenseInput = document.getElementById("expenseInput");
const categoryInput = document.getElementById("categoryInput");
const noteInput = document.getElementById("noteInput");

const monthIncome = document.getElementById("monthIncome");
const monthExpense = document.getElementById("monthExpense");
const monthSavings = document.getElementById("monthSavings");

function saveData() {
  localStorage.setItem("roseFinanceData", JSON.stringify(data));
}

function formatKey(year, month, day) {
  return `${year}-${month}-${day}`; // ✅ corrected
}

function generateCalendar() {
  calendar.innerHTML = "";

  let year = currentDate.getFullYear();
  let month = currentDate.getMonth();
  let daysInMonth = new Date(year, month + 1, 0).getDate();

  monthYear.innerText = `${currentDate.toLocaleString("default", { month: "long" })} ${year}`;

  for (let d = 1; d <= daysInMonth; d++) {
    let dayDiv = document.createElement("div");
    dayDiv.classList.add("day");
    dayDiv.innerText = d;

    let today = new Date();
    if (d === today.getDate() && month === today.getMonth() && year === today.getFullYear()) {
      dayDiv.classList.add("today");
    }

    let key = formatKey(year, month, d);
    if (data[key]) {
      let totalExpense = data[key].expenses?.reduce((sum, e) => sum + e.amount, 0) || 0;
      if (totalExpense > 0) {
        let badge = document.createElement("div");
        badge.classList.add("badge");
        badge.innerText = totalExpense;
        dayDiv.appendChild(badge);
      }
    }

    dayDiv.onclick = () => openModal(year, month, d);
    calendar.appendChild(dayDiv);
  }

  updateSummary();
}

function openModal(year, month, day) {
  selectedDay = formatKey(year, month, day);
  selectedDateText.innerText = `${day} ${monthYear.innerText}`; // ✅ corrected
  modal.style.display = "flex";
  loadEntries();
}

function loadEntries() {
  entryList.innerHTML = "";
  let entries = data[selectedDay]?.expenses || [];

  entries.forEach((entry, index) => {
    let div = document.createElement("div");
    div.classList.add("entry-item");
    div.innerHTML = `
      ₹${entry.amount} - ${entry.category} (${entry.note || ""})
      <button onclick="deleteEntry(${index})" style="float:right;background:red;color:white;border:none;border-radius:5px;">X</button>
    `;
    entryList.appendChild(div);
  });
}

function deleteEntry(index) {
  data[selectedDay].expenses.splice(index, 1);
  saveData();
  loadEntries();
  generateCalendar();
}

document.getElementById("saveEntry").onclick = () => {
  let income = parseFloat(incomeInput.value) || 0;
  let expense = parseFloat(expenseInput.value) || 0;

  if (!data[selectedDay]) {
    data[selectedDay] = { income: 0, expenses: [] };
  }

  if (income > 0) data[selectedDay].income = income;

  if (expense > 0) {
    data[selectedDay].expenses.push({
      amount: expense,
      category: categoryInput.value,
      note: noteInput.value
    });
  }

  incomeInput.value = "";
  expenseInput.value = "";
  noteInput.value = "";

  saveData();
  loadEntries();
  generateCalendar();
};

document.getElementById("closeModal").onclick = () => {
  modal.style.display = "none";
};

document.getElementById("prevMonth").onclick = () => {
  currentDate.setMonth(currentDate.getMonth() - 1);
  generateCalendar();
};

document.getElementById("nextMonth").onclick = () => {
  currentDate.setMonth(currentDate.getMonth() + 1);
  generateCalendar();
};

function updateSummary() {
  let year = currentDate.getFullYear();
  let month = currentDate.getMonth();
  let totalIncome = 0;
  let totalExpense = 0;

  Object.keys(data).forEach(key => {
    let [y, m] = key.split("-");
    if (parseInt(y) === year && parseInt(m) === month) {
      totalIncome += data[key].income || 0;
      totalExpense += data[key].expenses?.reduce((sum, e) => sum + e.amount, 0) || 0;
    }
  });

  monthIncome.innerText = totalIncome;
  monthExpense.innerText = totalExpense;
  monthSavings.innerText = totalIncome - totalExpense;
}

const themeToggle = document.getElementById("themeToggle");
let savedTheme = localStorage.getItem("theme") || "dark";
document.body.classList.add(savedTheme);

themeToggle.onclick = () => {
  document.body.classList.toggle("dark");
  document.body.classList.toggle("light");
  let mode = document.body.classList.contains("dark") ? "dark" : "light";
  localStorage.setItem("theme", mode);
};
